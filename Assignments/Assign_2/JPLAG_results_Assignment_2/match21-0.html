import numpy as np
import matplotlib.pyplot as plt
from numpy import linalg as LA
for n in range (1,50) :
nodes = []

V=np.zeros(shape=(n,n))
for i in range (0,n):
nodes.append(-1+(2*i+1)/n)
for i in range (0,n):
V[:i]=np.power(nodes,i)
Y=LA.cond(V)
plt.plot(n,Y)
plt.show();
