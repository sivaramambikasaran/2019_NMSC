% This Function Calculates Uniform Nodes
function xn=uni_nodes(n)
xn=zeros(n,1);
for k=1:n
xn(k)=-1+(2*(k-1)+1)/n;
endfor
endfunction

function [y] = lagrange(x, x0, y0)


n = size(x0, 1);
y = 0;

for i=1:n
p = 1;
for j=1:n

if j == i % avoiding fancy division by 0
continue;
endif

p =p.*(x-x0(j))./(x0(i)-x0(j));

endfor

y = y + y0(i) * p;
endfor
endfunction
